Table~\ref{tab:schedule} presents the project schedule organized into five phases across a 14-week development period spanning April 2026 through March 2027.

\begin{table}[htbp]
\centering
\renewcommand{\arraystretch}{1.4}
\caption{Project Schedule}
\label{tab:schedule}
\begin{tabular}{|p{3cm}|p{2cm}|p{2.5cm}|p{5.5cm}|}
\hline
\textbf{Phase} & \textbf{Weeks (Apr 2026)} & \textbf{Jan-Mar 2027} & \textbf{Tasks} \\
\hline
Planning \& Setup & W1-2 & Jan W1-2 & Study problem domain, prepare development tools/environment \\
\hline
Data Preparation & W3-5 & Jan W3-4 & Collect, label, organize ISU/YouTube video data \\
\hline
Feature Extraction & W1-2 & Feb W1-3 & MediaPipe pose extraction, 100-frame sequences \\
\hline
Model Development & W3-4 & Feb W4-Mar W2 & YOLO+MediaPipe integration, LSTM (Axel vs Non-Axel) \\
\hline
Extension \& Finalization & - & Mar W3-4 & Toe vs Edge extension, demo, final presentation \\
\hline
\end{tabular}
\end{table}

\begin{figure}[htbp]
\centering
\begin{tikzpicture}[scale=0.6, transform shape]  % Scale down to fit page
    \def\totalweeks{14}
    \def\barheight{0.4}
    \def\rowgap{0.7}

    % Week axis labels (match table exactly)
    \foreach \w in {1,3,6,9,12} {
        \node[font=\footnotesize, anchor=south] at (\w-0.5, 0.3) {\w};

    % Grid lines
    \foreach \w in {1,3,6,9,12,14} {
        \draw[gray!30, very thin] (\w-1, -0.2) -- (\w-1, -4.5);
    }

    % Bars - EXACTLY matching table weeks
    \foreach \i/\label/\start/\dur in { 
        0/Planning \& Setup/1/2,
        1/Data Preparation/3/3, 
        2/Feature Extraction/6/3,
        3/Model Development/9/3,
        4/Extension \& Finalization/12/3
    } {
        \pgfmathsetmacro\ypos{-\i * \rowgap - 0.5}
        % Bar
        \fill[blue!40, rounded corners=1pt]
            (\start-1, \ypos - \barheight/2) rectangle 
            (\start+\dur-1, \ypos + \barheight/2);
        % Label LEFT of chart
        \node[font=\footnotesize, anchor=east] at (-1.2, \ypos) {\label};
    }

    % Axis
    \draw[thick] (0, -0.2) -- (14, -0.2);
    \node[font=\small] at (7, 0.8) {Weeks};
