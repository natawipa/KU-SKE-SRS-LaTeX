\section{System Architecture Overview}
\label{section:architecture-overview}

FigureFlow processes broadcast video (1920$\times$1080@30fps) through a \textbf{three-stage pipeline} delivering jump predictions in \textbf{90ms/frame}.

\begin{figure}[htbp]
    \centering
    \includegraphics[width=0.8\textwidth]{images/system_overview.png}
    \caption{FigureFlow processing pipeline}
    \label{fig:system-overview}
\end{figure}

\label{fig:pipeline}
\end{figure}

\textbf{Stage Specifications}:
\begin{itemize}
\item \textbf{YOLOv8n}: 95\% mAP, 3.2M params, \textbf{30ms}
\item \textbf{MediaPipe}: 33 keypoints $\times$ 4 values/frame, \textbf{20ms}
\item \textbf{LSTM}: 100-frame sequences $\rightarrow$ jump type, \textbf{40ms}
\end{itemize}

\caption{Dataset specifications}

\textbf{ISU Labeling Pipeline}:
\begin{enumerate}
\item Parse PDFs: \texttt{3A<}, \texttt{2Lz!}, \texttt{3F+2T}
\item Match YouTube timestamps ($\pm$30s tolerance)
\item Extract $\pm$5s clips per jump
\item Validate via program sheets
\end{enumerate}

\textbf{Preprocessing}:
\begin{verbatim}
Video → YOLO crop (512×512) → 33×4 keypoints/frame 
     → Hip-normalize → 100-frame sequences → LSTM
\end{verbatim}

\textbf{Input Shape}: \textbf{33 $\times$ 100 $\times$ 4 = 13,200 features/jump}

\textbf{Training Configuration}:
\begin{itemize}
\item \textbf{Loss}: Binary cross-entropy (Axel weight=\textbf{3.2})
\item \textbf{Optimizer}: Adam (\texttt{lr=0.001})
\item \textbf{Split}: 80/20 (\textbf{200 train/50 test})
\item \textbf{Early stopping}: Patience=\textbf{10 epochs}
\end{itemize}
